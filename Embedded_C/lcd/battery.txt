void cw()
{

	rs=rw=0;
	map();
	toggle();
	B=B<<4;
	map();
	toggle();
}


void dw1()
{

	rs=1;
	rw=0;
	map();
	toggle();
	B=B<<4;
	map();
	toggle();
}


void toggle()
{
	e=1;
	_nop_();
	_nop_();
	_nop_();
	_nop_();
	e=0;
	_nop_();
	_nop_();
	_nop_();
}
